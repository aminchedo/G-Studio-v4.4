How to Avoid 429 Errors

The 429 Too Many Requests error occurs when you exceed API rate limits. Here's how to prevent it:

  • Implement a rate limiter to track requests per minute (RPM) and tokens per minute (TPM)
  • Use a queue system to manage concurrent requests
  • Add exponential backoff retry logic with jitter
  • Monitor Retry-After headers in responses
  • Respect the x-ratelimit-* headers returned by the API

interface RateLimiterConfig { requestsPerMinute: number; tokensPerMinute: number; maxConcurrent: number; retryAttempts: number; enableBackoff: boolean; } interface QueuedRequest { fn: () => Promise<any>; resolve: (value: any) => void; reject: (error: any) => void; estimatedTokens: number; } class GeminiRateLimiter { private config: RateLimiterConfig; private requestsThisMinute: number = 0; private tokensThisMinute: number = 0; private currentConcurrent: number = 0; private queue: QueuedRequest[] = []; private minuteResetTimer: NodeJS.Timeout | null = null; constructor(config: RateLimiterConfig) { this.config = config; this.startMinuteReset(); } private startMinuteReset(): void { this.minuteResetTimer = setInterval(() => { this.requestsThisMinute = 0; this.tokensThisMinute = 0; this.processQueue(); }, 60000); } async execute<T>( fn: () => Promise<T>, estimatedTokens: number = 100 ): Promise<T> { return new Promise((resolve, reject) => { this.queue.push({ fn, resolve, reject, estimatedTokens }); this.processQueue(); }); } private async processQueue(): Promise<void> { while ( this.queue.length > 0 && this.currentConcurrent < this.config.maxConcurrent ) { const item = this.queue[0]; if ( this.requestsThisMinute >= this.config.requestsPerMinute || this.tokensThisMinute + item.estimatedTokens > this.config.tokensPerMinute ) { break; } this.queue.shift(); this.currentConcurrent++; this.requestsThisMinute++; this.tokensThisMinute += item.estimatedTokens; this.executeWithRetry(item) .then(item.resolve) .catch(item.reject) .finally(() => { this.currentConcurrent--; this.processQueue(); }); } } private async executeWithRetry( item: QueuedRequest, attempt: number = 0 ): Promise<any> { try { return await item.fn(); } catch (error: any) { const is429 = error?.status === 429 || error?.message?.includes('429'); if ( is429 && this.config.enableBackoff && attempt < this.config.retryAttempts ) { const delay = Math.min( 1000 * Math.pow(2, attempt) + Math.random() * 1000, 60000 ); await new Promise(r => setTimeout(r, delay)); return this.executeWithRetry(item, attempt + 1); } throw error; } } getStatus() { return { requestsThisMinute: this.requestsThisMinute, tokensThisMinute: this.tokensThisMinute, queueSize: this.queue.length, currentConcurrent: this.currentConcurrent }; } destroy(): void { if (this.minuteResetTimer) { clearInterval(this.minuteResetTimer); } } }

VPN Impact on Gemini API Access

Using a VPN can affect your access to Google's Gemini API in several ways:

With VPN Active

  • Geo-restrictions: Some VPN IPs are blocked or rate-limited by Google
  • Increased latency: Extra routing through VPN servers adds 50-500ms delay
  • Shared IP issues: Other VPN users may have exhausted rate limits on that IP
  • Suspicious activity detection: Rapid IP changes can trigger security blocks
  • Country-specific models: Some models may only be available in certain regions

Recommendation: If testing fails with VPN, try disabling it temporarily or use a premium VPN with dedicated IPs.

Without VPN

  • Direct access: Fastest connection to Google's servers
  • Lower latency: Typical response times 100-300ms
  • Clean IP reputation: Your ISP IP is less likely to be rate-limited
  • Consistent access: More reliable API responses
  • Full model availability: Access to all region-specific models

Recommendation: For production use, avoid VPNs unless required for compliance.

Configuration Adjustments for VPN

If you must use a VPN, adjust these settings:

  • Increase timeouts: Set fetch timeout to 30-60 seconds instead of 10
  • Reduce concurrent requests: Lower from 5 to 1-2
  • Add delay between requests: Wait 2-5 seconds between calls
  • Enable aggressive retry: Retry up to 5 times with exponential backoff
  • Monitor specific errors: Watch for ECONNREFUSED, ETIMEDOUT, 403 Forbidden

interface FetchConfig { timeout: number; retries: number; vpnMode: boolean; } async function fetchWithVPNSupport( url: string, options: RequestInit, config: FetchConfig ): Promise<Response> { const timeout = config.vpnMode ? 60000 : config.timeout; const maxRetries = config.vpnMode ? 5 : config.retries; for (let attempt = 0; attempt <= maxRetries; attempt++) { try { const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), timeout); const response = await fetch(url, { ...options, signal: controller.signal }); clearTimeout(timeoutId); if (response.ok || (response.status >= 400 && response.status < 500 && response.status !== 429)) { return response; } const baseDelay = config.vpnMode ? 3000 : 1000; const delay = Math.min( baseDelay * Math.pow(2, attempt) + Math.random() * 1000, 60000 ); if (attempt < maxRetries) { console.warn(`Retry ${attempt + 1}/${maxRetries} after ${delay}ms`); await new Promise(r => setTimeout(r, delay)); } else { return response; } } catch (error: any) { if (attempt === maxRetries) throw error; const isTimeout = error.name === 'AbortError'; const isNetwork = error.message?.includes('network') || error.message?.includes('fetch'); if (isTimeout || isNetwork) { const delay = 3000 * Math.pow(2, attempt); console.warn(`Network error, retry ${attempt + 1}/${maxRetries} after ${delay}ms`); await new Promise(r => setTimeout(r, delay)); } else { throw error; } } } throw new Error('Max retries exceeded'); }
