3. Rate Limiter Class

utils/rate-limiter.ts 
export class RateLimiter {
  private requestsPerMinute = 15;
  private tokensPerMinute = 1000000;
  private requestCount = 0;
  private tokenCount = 0;
  private resetTime = Date.now() + 60000;

  async executeWithLimit(
    fn: () => Promise,
    estimatedTokens: number = 1000
  ): Promise {
    // Reset if needed
    if (Date.now() >= this.resetTime) {
      this.requestCount = 0;
      this.tokenCount = 0;
      this.resetTime = Date.now() + 60000;
    }

    // Check limits
    if (this.requestCount >= this.requestsPerMinute) {
      const waitTime = this.resetTime - Date.now();
      console.log(`⏳ Rate limit. Waiting ${waitTime}ms...`);
      await this.sleep(waitTime);
      return this.executeWithLimit(fn, estimatedTokens);
    }

    if (this.tokenCount + estimatedTokens > this.tokensPerMinute) {
      const waitTime = this.resetTime - Date.now();
      console.log(`⏳ Token limit. Waiting ${waitTime}ms...`);
      await this.sleep(waitTime);
      return this.executeWithLimit(fn, estimatedTokens);
    }

    // Execute with retry
    let attempt = 0;
    const maxAttempts = 5;

    while (attempt < maxAttempts) {
      try {
        this.requestCount++;
        this.tokenCount += estimatedTokens;
        return await fn();
      } catch (error: any) {
        if (error.status === 429) {
          attempt++;
          const waitTime = Math.pow(2, attempt) * 1000;
          console.log(`⚠️ 429. Retry ${attempt}/${maxAttempts}`);
          await this.sleep(waitTime);
        } else {
          throw error;
        }
      }
    }

    throw new Error(`Failed after ${maxAttempts} attempts`);
  }

  private sleep(ms: number): Promise {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

4. Gemini Client

client/gemini-client.ts 
import { RateLimiter } from '../utils/rate-limiter';
import { GeminiModel, GenerateContentRequest } from '../types/gemini';

export class GeminiClient {
  private apiKey: string;
  private baseUrl = 'https://generativelanguage.googleapis.com/v1beta';
  private rateLimiter: RateLimiter;

  constructor(apiKey: string) {
    if (!apiKey) throw new Error('API key required');
    this.apiKey = apiKey;
    this.rateLimiter = new RateLimiter();
  }

  async listModels(): Promise {
    const response = await this.fetchWithRetry(
      `${this.baseUrl}/models?key=${this.apiKey}`
    );
    const data = await response.json();
    return data.models || [];
  }

  async generateContent(
    model: string,
    request: GenerateContentRequest,
    estimatedTokens = 1000
  ) {
    return this.rateLimiter.executeWithLimit(async () => {
      const response = await this.fetchWithRetry(
        `${this.baseUrl}/${model}:generateContent?key=${this.apiKey}`,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(request)
        }
      );
      return response.json();
    }, estimatedTokens);
  }

  private async fetchWithRetry(
    url: string,
    options?: RequestInit,
    maxAttempts = 5
  ): Promise {
    let attempt = 0;

    while (attempt < maxAttempts) {
      try {
        const response = await fetch(url, options);

        if (response.status === 429) {
          attempt++;
          const waitTime = Math.pow(2, attempt) * 1000;
          await new Promise(r => setTimeout(r, waitTime));
          continue;
        }

        if (!response.ok) {
          throw new Error(`HTTP ${response.status}`);
        }

        return response;
      } catch (error) {
        if (attempt === maxAttempts - 1) throw error;
        attempt++;
      }
    }

    throw new Error(`Failed after ${maxAttempts} attempts`);
  }
}

How VPN Affects Gemini API Access

با VPN (With VPN)

  • ⚠️ Google ممکنه برخی مدل‌ها رو محدود یا بلاک کنه
  • ⚠️ احتمال دریافت 403 Forbidden یا 429 Too Many Requests
  • ✗ برخی کشورها (مثل ایران) به صورت کامل بلاک هستن
  • ⚠️ VPN های رایگان معمولاً مشکل بیشتری دارن
  • ⏱️ Latency بالاتر — پاسخ کندتر میشه

بدون VPN (Without VPN)

  • ✓ دسترسی مستقیم و سریع‌تر
  • ✓ احتمال 429 کمتر (با رعایت rate limit)
  • ✓ Latency پایین‌تر
  • ✗ در برخی کشورها (مثل ایران) بدون VPN دسترسی نداری
راه‌حل‌های پیشنهادی
  • از VPN های premium استفاده کن (ExpressVPN, NordVPN)
  • سرور VPN رو به کشورهای دوستانه با Google تغییر بده (آمریکا، کانادا، آلمان)
  • اگه میشه، از Proxy های مخصوص API استفاده کن
  • Rate Limiting هوشمند رو فعال کن (این ابزار خودکار این کار رو میکنه)
  • از Retry-After header توی پاسخ 429 استفاده کن

تشخیص VPN در این ابزار

این ابزار به صورت خودکار IP و موقعیت جغرافیایی شما رو تشخیص میده، اگه از VPN استفاده میکنی بهت نشون میده، و Latency رو اندازه‌گیری میکنه تا ببینی چقدر تأخیر داری.

نکته مهم
Google AI Studio (Gemini) در برخی کشورها رسماً در دسترس نیست. لیست کامل کشورهای پشتیبانی‌شده رو از این لینک ببین.

1️⃣ Request Queuing (صف درخواست)

درخواست‌ها رو توی یه صف بریز و به صورت تک‌تک با delay پردازش کن:

JavaScript Example 
const queue = [];
const delay = 4000; // 4 seconds between requests

async function processQueue() {
  while (queue.length > 0) {
    const request = queue.shift();
    await request();
    await new Promise(r => setTimeout(r, delay));
  }
}

2️⃣ Exponential Backoff (تأخیر تصاعدی)

وقتی 429 گرفتی، به صورت تصاعدی منتظر بمون: 1s → 2s → 4s → 8s → 16s

JavaScript Example 
async function retryWithBackoff(fn, maxAttempts = 5) {
  for (let i = 0; i < maxAttempts; i++) {
    try {
      return await fn();
    } catch (error) {
      if (error.status === 429 && i < maxAttempts - 1) {
        const waitTime = Math.pow(2, i) * 1000;
        await new Promise(r => setTimeout(r, waitTime));
      } else {
        throw error;
      }
    }
  }
}

4️⃣ Concurrent Limiting (محدودیت همزمانی)

حداکثر 2-3 درخواست همزمان بزن:

JavaScript Example 
async function limitConcurrency(tasks, limit = 3) {
  const results = [];
  const executing = [];

  for (const task of tasks) {
    const promise = task().then(result => {
      executing.splice(executing.indexOf(promise), 1);
      return result;
    });

    results.push(promise);
    executing.push(promise);

    if (executing.length >= limit) {
      await Promise.race(executing);
    }
  }

  return Promise.all(results);
}
