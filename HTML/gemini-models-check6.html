3. Rate Limiter Class (utils/rate-limiter.ts)

export class RateLimiter { private requestsPerMinute = 15; private tokensPerMinute = 1000000; private requestCount = 0; private tokenCount = 0; private resetTime = Date.now() + 60000; async executeWithLimit( fn: () => Promise, estimatedTokens: number = 1000 ): Promise { // Reset if needed if (Date.now() >= this.resetTime) { this.requestCount = 0; this.tokenCount = 0; this.resetTime = Date.now() + 60000; } // Check request limit if (this.requestCount >= this.requestsPerMinute) { const waitTime = this.resetTime - Date.now(); console.log(`‚è≥ Rate limit reached. Waiting ${waitTime}ms...`); await this.sleep(waitTime); return this.executeWithLimit(fn, estimatedTokens); } // Check token limit if (this.tokenCount + estimatedTokens > this.tokensPerMinute) { const waitTime = this.resetTime - Date.now(); console.log(`‚è≥ Token limit reached. Waiting ${waitTime}ms...`); await this.sleep(waitTime); return this.executeWithLimit(fn, estimatedTokens); } // Execute with retry let attempt = 0; const maxAttempts = 5; while (attempt < maxAttempts) { try { this.requestCount++; this.tokenCount += estimatedTokens; return await fn(); } catch (error: any) { if (error.status === 429) { attempt++; const retryAfter = error.headers?.get('Retry-After'); const waitTime = retryAfter ? parseInt(retryAfter) * 1000 : Math.pow(2, attempt) * 1000; console.log(`‚ö†Ô∏è 429 Error. Retry ${attempt}/${maxAttempts} after ${waitTime}ms`); await this.sleep(waitTime); } else { throw error; } } } throw new Error(`Failed after ${maxAttempts} attempts`); } private sleep(ms: number): Promise { return new Promise(resolve => setTimeout(resolve, ms)); } getStatus() { return { requestCount: this.requestCount, tokenCount: this.tokenCount, requestsPerMinute: this.requestsPerMinute, tokensPerMinute: this.tokensPerMinute, resetTime: this.resetTime }; } }

2Ô∏è‚É£ Exponential Backoff

When you get a 429, wait progressively longer before retrying:

async function retryWithBackoff(fn, maxAttempts = 5) { for (let i = 0; i < maxAttempts; i++) { try { return await fn(); } catch (error) { if (error.status === 429 && i < maxAttempts - 1) { const waitTime = Math.pow(2, i) * 1000; // 1s, 2s, 4s, 8s, 16s await new Promise(r => setTimeout(r, waitTime)); } else { throw error; } } } }

4Ô∏è‚É£ Concurrent Request Limiting

Limit the number of simultaneous requests:

async function limitConcurrency(tasks, limit = 3) { const results = []; const executing = []; for (const task of tasks) { const promise = task().then(result => { executing.splice(executing.indexOf(promise), 1); return result; }); results.push(promise); executing.push(promise); if (executing.length >= limit) { await Promise.race(executing); } } return Promise.all(results); }

üåê VPN Considerations

With VPN:
‚Ä¢ Google may block or limit certain models
‚Ä¢ Higher chance of 403 or 429 errors
‚Ä¢ Some countries (like Iran) are completely blocked
‚Ä¢ Free VPNs often cause more issues
‚Ä¢ Higher latency = slower responses

Without VPN:
‚Ä¢ Direct, faster access
‚Ä¢ Lower chance of 429 (if you respect rate limits)
‚Ä¢ Lower latency
‚Ä¢ But: In some countries, you won't have access at all

Recommendations:
‚Ä¢ Use premium VPNs (ExpressVPN, NordVPN)
‚Ä¢ Choose servers in Google-friendly countries (US, Canada, Germany)
‚Ä¢ Use API-specific proxies if possible
‚Ä¢ Enable intelligent rate limiting (this tool does it automatically)
